% Examples of how to do figures and tables.

%
% Normal figure.
%
\begin{figure}[t!]
\centering
\includegraphics[scale=1.0]{graphics/fig-xxxxx.pdf}
\caption{Figure caption goes here.}
\label{fig-xxxxx}
\end{figure}
